 1 def run():
 2     import signal
 3     from os import fspath

 7     from PySide6.QtQml import QQmlApplicationEngine

14     QQuickStyle.setStyle("Universal")

18     engine.addImportPath(fspath(Path(__file__).parent))
19     engine.loadFromModule("HowTo.Ctors", "Main")
20 
21     exit_code = app.exec()
